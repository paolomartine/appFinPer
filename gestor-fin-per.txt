    <select id="tipo">

    </select>

    <button onclick="agregar()">Agregar</button>

<script>
    let transacciones = JSON.parse(localStorage.getItem("transacciones")) || [];

    function guardar() {
        localStorage.setItem("transacciones", JSON.stringify(transacciones));
    }

    function render() {
        let lista = document.getElementById("lista");
        lista.innerHTML = "";

        let ingresos = 0;
        let gastos = 0;

        transacciones.forEach((t, index) => {
            let li = document.createElement("li");
            li.className = t.tipo;

            li.innerHTML = `
                ${t.descripcion} - $${t.monto}
                <button onclick="eliminar(${index})">X</button>
            `;

            lista.appendChild(li);

            if (t.tipo === "ingreso") ingresos += t.monto;
            else gastos += t.monto;
        });

        document.getElementById("ingresos").innerText = ingresos;
        document.getElementById("gastos").innerText = gastos;
        document.getElementById("balance").innerText = ingresos - gastos;
    }

    function agregar() {
        let descripcion = document.getElementById("descripcion").value;
        let monto = parseFloat(document.getElementById("monto").value);
        let tipo = document.getElementById("tipo").value;

        if (!descripcion || isNaN(monto)) {
            alert("Completa todos los campos");
            return;
        }

        transacciones.push({ descripcion, monto, tipo });
        guardar();
        render();

        document.getElementById("descripcion").value = "";
        document.getElementById("monto").value = "";
    }

    function eliminar(index) {
        transacciones.splice(index, 1);
        guardar();
        render();
    }

    render();
</script>
